%n-dimensional volume of a ball
\text{The n-dimensional volume of a ball of radius r is } \\
$\displaystyle V_{n}(r)={\frac {\pi ^{n/2}}{\Gamma {\bigl (}{\tfrac {n}{2}}+1{\bigr )}}}r^{n}$\\
$\text{where } \Gamma(n) = (n - 1)! \text { and } \Gamma(n + \frac{1}{2}) = (n - \frac{1}{2}) \cdot \ldots \cdot \frac{1}{2} \cdot \pi^{\frac{1}{2}} $ \\[2ex]

%Rodrigues' rotation formula
$\text{If } \mathbf {v} \in \mathbb{R}^3 \text{ and } k \text{ is a unit vector describing an axis }$ \\
$\text{of rotation about which } \mathbf {v} \text{ rotates by an angle } \theta$ \\ 
$\text{ according to the right hand rule:}$ \\
$\displaystyle \mathbf {v} _{\mathrm {rot} }=\mathbf {v} \cos \theta +(\mathbf {k} \times \mathbf {v} )\sin \theta +\mathbf {k} ~(\mathbf {k} \cdot \mathbf {v} )(1-\cos \theta )$ \\[2ex]

%Spherical cones
$\text{Spherical cone with sphere radius r,}$\\
$\text{height h and radius of circle a:}$\\
$A=\pi r(2h+a)$ \text{(includes lateral area)}\\
$V=\frac{2}{3}\pi r^2h$\\[2ex]

%Pappus
\text{Surface/Solid of revolution} \\
$\displaystyle A=2\pi Ld$, $L = \text{length of curve}$ \\
$\displaystyle V=2\pi Ad$, $A = \text{area of surface}$ \\
$d = \text{distance of centroid to axis}$ \\
